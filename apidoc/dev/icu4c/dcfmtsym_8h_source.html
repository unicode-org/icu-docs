544inline const UnicodeString& DecimalFormatSymbols::getConstDigitSymbol(int32_t digit) const {
545 if (digit < 0 || digit > 9) {
546 digit = 0;
547 }
548 if (digit == 0) {
549 return fSymbols[kZeroDigitSymbol];
550 }
551 ENumberFormatSymbol key = static_cast<ENumberFormatSymbol>(kOneDigitSymbol + digit - 1);
552 return fSymbols[key];
553}

558inline void
559DecimalFormatSymbols::setSymbol(ENumberFormatSymbol symbol, const UnicodeString &value, const UBool propagateDigits = true) {
560 if (symbol == kCurrencySymbol) {
561 fIsCustomCurrencySymbol = true;
562 }
563 else if (symbol == kIntlCurrencySymbol) {
564 fIsCustomIntlCurrencySymbol = true;
565 }
566 if(symbol<kFormatSymbolCount) {
567 fSymbols[symbol]=value;
568 }
569
570 // If the zero digit is being set to a known zero digit according to Unicode,
571 // then we automatically set the corresponding 1-9 digits
572 // Also record updates to fCodePointZero. Be conservative if in doubt.
573 if (symbol == kZeroDigitSymbol) {
574 UChar32 sym = value.char32At(0);
575 if ( propagateDigits && u_charDigitValue(sym) == 0 && value.countChar32() == 1 ) {
576 fCodePointZero = sym;
577 for ( int8_t i = 1 ; i<= 9 ; i++ ) {
578 sym++;
579 fSymbols[(int)kOneDigitSymbol+i-1] = UnicodeString(sym);
580 }
581 } else {
582 fCodePointZero = -1;
583 }
584 } else if (symbol >= kOneDigitSymbol && symbol <= kNineDigitSymbol) {
585 fCodePointZero = -1;
586 }
587}
