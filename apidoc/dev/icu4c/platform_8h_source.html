645#ifdef U_SIZEOF_WCHAR_T
646 /* Use the predefined value. */
647#elif (U_PLATFORM == U_PF_ANDROID && __ANDROID_API__ < 9)
648 /*
649 * Classic Mac OS and Mac OS X before 10.3 (Panther) did not support wchar_t or wstring.
650 * Newer Mac OS X has size 4.
651 */
652# define U_SIZEOF_WCHAR_T 1
653#elif U_PLATFORM_HAS_WIN32_API || U_PLATFORM == U_PF_CYGWIN
654# define U_SIZEOF_WCHAR_T 2
655#elif U_PLATFORM == U_PF_AIX
656 /*
657 * AIX 6.1 information, section "Wide character data representation":
658 * "... the wchar_t datatype is 32-bit in the 64-bit environment and
659 * 16-bit in the 32-bit environment."
660 * and
661 * "All locales use Unicode for their wide character code values (process code),
662 * except the IBM-eucTW codeset."
663 */
664# ifdef __64BIT__
665# define U_SIZEOF_WCHAR_T 4
666# else
667# define U_SIZEOF_WCHAR_T 2
668# endif
669#elif U_PLATFORM == U_PF_OS390
670 /*
671 * z/OS V1R11 information center, section "LP64 | ILP32":
672 * "In 31-bit mode, the size of long and pointers is 4 bytes and the size of wchar_t is 2 bytes.
673 * Under LP64, the size of long and pointer is 8 bytes and the size of wchar_t is 4 bytes."
674 */
675# ifdef _LP64
676# define U_SIZEOF_WCHAR_T 4
677# else
678# define U_SIZEOF_WCHAR_T 2
679# endif
680#elif U_PLATFORM == U_PF_OS400
681# if defined(__UTF32__)
682 /*
683 * LOCALETYPE(*LOCALEUTF) is specified.
684 * Wide-character strings are in UTF-32,
685 * narrow-character strings are in UTF-8.
686 */
687# define U_SIZEOF_WCHAR_T 4
688# elif defined(__UCS2__)
689 /*
690 * LOCALETYPE(*LOCALEUCS2) is specified.
691 * Wide-character strings are in UCS-2,
692 * narrow-character strings are in EBCDIC.
693 */
694# define U_SIZEOF_WCHAR_T 2
695# else
696 /*
697 * LOCALETYPE(*CLD) or LOCALETYPE(*LOCALE) is specified.
698 * Wide-character strings are in 16-bit EBCDIC,
699 * narrow-character strings are in EBCDIC.
700 */
701# define U_SIZEOF_WCHAR_T 2
702# endif
703#else
704# define U_SIZEOF_WCHAR_T 4
705#endif
