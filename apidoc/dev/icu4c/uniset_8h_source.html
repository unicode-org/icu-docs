1756inline int32_t UnicodeSet::span(const UnicodeString &s, int32_t start, USetSpanCondition spanCondition) const {
1757 int32_t sLength=s.length();
1758 if(start<0) {
1759 start=0;
1760 } else if(start>sLength) {
1761 start=sLength;
1762 }
1763 return start+span(s.getBuffer()+start, sLength-start, spanCondition);
1764}
