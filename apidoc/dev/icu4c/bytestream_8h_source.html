9 // Abstract interface that consumes a sequence of bytes (ByteSink).
10 //
11 // Used so that we can write a single piece of code that can operate
12 // on a variety of output string types.
13 //
14 // Various implementations of this interface are provided:
15 // ByteSink:
16 // CheckedArrayByteSink Write to a flat array, with bounds checking
17 // StringByteSink Write to an STL string

20 // a compromise between the original Google code and the ICU coding guidelines.
21 // For example, data types are ICU-ified (size_t,int->int32_t),
22 // and API comments doxygen-ified, but function names and behavior are
23 // as in the original, if possible.

268 template<typename StringClass>
269 class StringByteSink : public ByteSink {
270  using Unit = typename StringClass::value_type;
271  public:
277  StringByteSink(StringClass* dest) : dest_(dest) { }
285  StringByteSink(StringClass* dest, int32_t initialAppendCapacity) : dest_(dest) {
286  if (initialAppendCapacity > 0 &&
287  static_cast<uint32_t>(initialAppendCapacity) > dest->capacity() - dest->length()) {
288  dest->reserve(dest->length() + initialAppendCapacity);
289  }
290  }
297  virtual void Append(const char* data, int32_t n) override {
298  if constexpr (std::is_same_v<Unit, char>) {
299  dest_->append(data, n);
300  } else {
301  dest_->append(reinterpret_cast<const Unit*>(data), n);
302  }
303  }
304  private:
305  StringClass* dest_;
306 
307  StringByteSink() = delete;
308  StringByteSink(const StringByteSink &) = delete;
309  StringByteSink &operator=(const StringByteSink &) = delete;
